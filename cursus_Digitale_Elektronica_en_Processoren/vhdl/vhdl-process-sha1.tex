library ieee;
use ieee.std_logic_1164.all;

entity shacalc is
    port (
        clk : in std_logic;
        data : in std_logic_vector(511 downto 0);
        enable : in std_logic;
        signature : out std_logic_vector (159 downto 0);
        done : out std_logic;
    );
end entity shacalc;

architecture prototype of shacalc
    signal sFlag: std_logic;
    signal sD: std_logic;
    signal sQ: std_logic;
begin
    with sFlag select
	  sQ <= sD  when '1',
		'0' when others;
end architecture prototype;
